## 修改建议

### 1. 清理脚本安全增强

```bash
# 在 cleanup-project.sh 开头添加
validate_project_root() {
    local root="$1"
    # 禁止在根目录、家目录执行
    if [[ "$root" == "/" || "$root" == "$HOME" || "$root" == ~* ]]; then
        echo "错误：不允许在系统目录或家目录执行清理"
        exit 1
    fi
    # 必须有 .eket 目录才认为是有效项目
    if [[ ! -d "$root/.eket" ]]; then
        echo "错误：未找到有效的 EKET 项目结构"
        exit 1
    fi
}
```

### 2. 版本文件

```yaml
# .eket/version
version: 0.1.0
created_at: 2026-03-19
template_version: 0.1.0
```

### 3. 健康检查脚本

```bash
#!/bin/bash
# .eket/health_check.sh

check_status() {
    echo "检查 EKET 项目健康状态..."

    # 检查必要文件
    local required_files=(
        "CLAUDE.md"
        ".eket/config.yml"
        "inbox/human_input.md"
    )

    for file in "${required_files[@]}"; do
        if [[ -f "$file" ]]; then
            echo "✓ $file"
        else
            echo "✗ $file (缺失)"
        fi
    done

    # 检查目录
    local required_dirs=(
        ".claude/commands"
        "inbox/human_feedback"
        "outbox/review_requests"
        "tasks"
    )

    for dir in "${required_dirs[@]}"; do
        if [[ -d "$dir" ]]; then
            echo "✓ $dir"
        else
            echo "✗ $dir (缺失)"
        fi
    done
}

check_status
```
