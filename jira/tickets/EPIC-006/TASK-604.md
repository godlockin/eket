# TASK-604: Smart File Reader（分段 + 摘要）

**EPIC**: EPIC-006 | **Milestone**: M1-Optimization | **优先级**: P1 | **工时**: 6h | **状态**: ready | **依赖**: TASK-603

## 需求

实现智能文件读取：<10KB 直接读，10-50KB 读取并生成摘要，>50KB 仅读前 500 行 + 结构分析。

## 验收标准

- **AC-1**: Given Read 8KB 文件, When `smartRead()` 调用, Then 返回完整内容（无摘要）
- **AC-2**: Given Read 30KB 文件, When `smartRead()` 调用, Then 返回：文件头 + Haiku 生成的 200 字摘要 + "Full content available via explicit request" 提示
- **AC-3**: Given Read 100KB 文件, When `smartRead()` 调用, Then 返回：前 500 行 + 结构分析（函数签名、类定义、import 列表）+ line range 提示
- **AC-4**: Given 摘要生成调用 Haiku, When 统计成本, Then 单次摘要 < 0.01 USD（Haiku input ~2k tokens @ $0.25/MTok）
- **AC-5**: Given Slaver 使用 smartRead 分析 5 个大文件, When 对比传统 Read, Then context 节省 > 60%

### 核心实现
```typescript
// node/src/core/smart-reader.ts
import * as fs from 'fs/promises';
import { contextTracker } from './context-tracker.js';
import { summarizeWithHaiku } from './model-provider.js';

export async function smartRead(filePath: string, sessionId: string): Promise<string> {
  const stats = await fs.stat(filePath);
  const sizeKB = stats.size / 1024;

  // 策略 1: 小文件（<10KB）直接读
  if (sizeKB < 10) {
    const content = await fs.readFile(filePath, 'utf-8');
    contextTracker.trackToolOutput(sessionId, content);
    return content;
  }

  // 策略 2: 中文件（10-50KB）读取 + 摘要
  if (sizeKB < 50) {
    const content = await fs.readFile(filePath, 'utf-8');
    const summary = await summarizeWithHaiku(content.slice(0, 10000));
    
    const output = `[File: ${filePath}, ${sizeKB.toFixed(1)} KB]

${content.slice(0, 2000)}

--- Summary (generated by Haiku) ---
${summary}

--- Full content available ---
Use: Read tool with full path to see complete content
`;
    
    contextTracker.trackToolOutput(sessionId, output);
    return output;
  }

  // 策略 3: 大文件（>50KB）前 500 行 + 结构
  const content = await fs.readFile(filePath, 'utf-8');
  const lines = content.split('\n');
  const preview = lines.slice(0, 500).join('\n');
  const structure = await analyzeFileStructure(content, filePath);

  const output = `[File: ${filePath}, ${lines.length} lines, ${sizeKB.toFixed(1)} KB]

--- Preview (first 500 lines) ---
${preview}

--- File Structure ---
${structure}

--- Read with line range ---
Use: Read tool with offset/limit to see specific sections
Example: Read ${filePath} --offset 500 --limit 500
`;

  contextTracker.trackToolOutput(sessionId, output);
  return output;
}

async function summarizeWithHaiku(content: string): Promise<string> {
  // 调用 Haiku 模型生成摘要（成本低）
  const result = await runClaude({
    prompt: `Summarize this code in 200 words, focus on: purpose, key functions, dependencies, patterns.\n\n${content}`,
    projectRoot: process.cwd(),
    model: 'haiku',
    role: 'summarizer',
  });

  return result.stdout || '[Summary generation failed]';
}
```

```typescript
// node/src/core/file-structure-analyzer.ts
export async function analyzeFileStructure(content: string, filePath: string): Promise<string> {
  const ext = path.extname(filePath);
  
  switch (ext) {
    case '.ts':
    case '.js':
      return analyzeJavaScriptStructure(content);
    case '.py':
      return analyzePythonStructure(content);
    case '.rs':
      return analyzeRustStructure(content);
    default:
      return 'Structure analysis not available for this file type';
  }
}

function analyzeJavaScriptStructure(content: string): string {
  const imports = content.match(/^import .+ from .+$/gm) || [];
  const exports = content.match(/^export (function|class|const|interface) \w+/gm) || [];
  const functions = content.match(/^(export )?function \w+/gm) || [];
  const classes = content.match(/^(export )?class \w+/gm) || [];

  return `
Imports (${imports.length}):
${imports.slice(0, 10).join('\n')}

Exports (${exports.length}):
${exports.join('\n')}

Functions (${functions.length}):
${functions.join('\n')}

Classes (${classes.length}):
${classes.join('\n')}
`.trim();
}
```
